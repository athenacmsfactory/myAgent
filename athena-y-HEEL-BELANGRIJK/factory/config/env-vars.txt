# Athena CMS - Vereiste Omgevingsvariabelen (.env)

Hieronder volgt een overzicht van alle variabelen die zijn geïdentificeerd in de Athena CMS codebase. Gebruik dit om uw .env bestand te herstellen.

## 🔑 API Keys
GEMINI_API_KEY=          # Uw Google Gemini API key (vereist voor AI parsing)
GOOGLE_SHEETS_API_KEY=   # Uw Google Sheets API key (voor publieke sheet toegang)

## 🌐 GitHub Configuratie (voor Deployment Wizard)
GITHUB_USER=             # Uw GitHub gebruikersnaam
GITHUB_PAT=              # Uw GitHub Personal Access Token (met repo scopes)
GITHUB_ORG=              # (Optioneel) GitHub organisatie naam (default: GITHUB_USER)
GITHUB_SSH_HOST=         # (Optioneel) SSH Host alias (default: github.com)
GH_TOKEN=                # (Vaak gelijk aan GITHUB_PAT, gebruikt door gh-cli)

## 📁 Systeempaden
ROOT_DIR=                # Absolute pad naar de project root (bv. /home/kareltestspecial/athena/factory)
ATHENA_ROOT=             # Vaak gelijk aan ROOT_DIR

## 📊 Google Sheets & Drive (voor Auto-Provisioning & OAuth)
MASTER_TEMPLATE_ID=      # ID van de Master Template Google Sheet
DRIVE_PROJECTS_FOLDER_ID= # ID van de Google Drive map voor nieuwe projecten
GOOGLE_CLIENT_ID=        # OAuth2 Client ID
GOOGLE_CLIENT_SECRET=    # OAuth2 Client Secret
GOOGLE_REFRESH_TOKEN=    # OAuth2 Refresh Token voor offline toegang
GOOGLE_SERVICE_ACCOUNT_EMAIL= # Email van het service account voor Sheet toegang

## 🤖 AI Model Configuratie (Optioneel - fallbacks aanwezig)
AI_MODEL_DATA_ARCHITECT=  # Default: gemini-flash-latest
AI_MODEL_SITETYPE_PROMPT_ENGINEER= # Default: gemini-pro-latest
AI_MODEL_PARSER=          # Default: gemini-flash-latest
AI_MODEL_IMAGE_GENERATOR= # Default: gemini-flash-latest
AI_MODEL_DEFAULT=         # Algemene fallback: gemini-flash-latest

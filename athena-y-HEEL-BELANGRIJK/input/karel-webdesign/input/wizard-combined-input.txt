const DEFAULT_LINKS: NavLink[] = [

  { name: 'Diensten', href: '#services' },

const Navbar: React.FC = () => {
  const [navLinks, setNavLinks] = useState<NavLink[]>(DEFAULT_LINKS);

  useEffect(() => {
    sanityClient
      .fetch(
        `*[_type == "navigationLink"] | order(order asc) {
          name,
          href
        }`
      )
      .then((data) => {
        if (data && data.length > 0) {
          setNavLinks(data);
        }
      })
      .catch((err) => {
        console.error('Failed to fetch navigation links:', err);
        // Fallback is already set in initial state
      });
  }, []);

  const handleScroll = (e: React.MouseEvent<HTMLAnchorElement>, href: string) => {
    e.preventDefault();
    const targetId = href.substring(1);
    const targetElement = document.getElementById(targetId);
    if (targetElement) {
      targetElement.scrollIntoView({ behavior: 'smooth' });
    }
  };

  return (
    <header className="fixed top-0 left-0 right-0 z-50 bg-white/90 backdrop-blur-md shadow-sm">
      <nav className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8" aria-label="Top">
        <div className="w-full py-6 flex items-center justify-between border-b border-slate-200 border-opacity-20 md:border-none">
          <div className="flex items-center">
            <a href="#" onClick={(e) => handleScroll(e, '#')} className="text-slate-900 text-lg font-bold">
              Karel Webdesign
            </a>
          </div>
          <div className="hidden ml-10 space-x-8 md:block">
            {navLinks.map((link) => (
              <a
                key={link.name}
                href={link.href}
                onClick={(e) => handleScroll(e, link.href)}
                className="text-base font-medium text-slate-500 hover:text-slate-900"
              >
                {link.name}
              </a>
            ))}
          </div>
        </div>
        <div className="py-4 px-4 flex flex-nowrap overflow-x-auto justify-start space-x-4 md:hidden">
          {navLinks.map((link) => (
            <a 
              key={link.name}
              href={link.href}
              onClick={(e) => handleScroll(e, link.href)}
              className="text-base font-medium text-slate-500 hover:text-slate-900"
            >
              {link.name}
            </a>
          ))}
        </div>
      </nav>
    </header>
  );
};

const builder = createImageUrlBuilder(sanityClient);

function urlFor(source: any) {
  try {
    return builder.image(source);
  } catch (e) {
    return null;
  }
}

function generateSlug(text: string) {
  return text
    .toLowerCase()
    .replace(/\s+/g, '-')
    .replace(/[^\w-]+/g, '');
}

const PortfolioImage: React.FC<{ item: PortfolioItem }> = ({ item }) => {
  const [imgSrc, setImgSrc] = useState<string | null>(null);

  useEffect(() => {
    // 1. Probeer eerst Sanity URL
    if (item.imageUrl) {
      const sanityUrl = urlFor(item.imageUrl)?.width(800).url();
      if (sanityUrl) {
        setImgSrc(sanityUrl);
        return;
      }
    }
    // 2. Als geen Sanity image, val direct terug op lokaal
    setImgSrc(`/images/portfolio/${generateSlug(item.title)}.jpg`);
  }, [item]);

  const handleError = () => {
    // 3. Als laden faalt, fallback naar lokaal (als we dat nog niet deden)
    const localUrl = `/images/portfolio/${generateSlug(item.title)}.jpg`;
    if (imgSrc !== localUrl) {
      setImgSrc(localUrl);
    }
  };

  return (
    <img 
      className="h-full w-full object-cover transition-transform duration-500 group-hover:scale-105" 
      src={imgSrc || ''} 
      alt={item.title}
      onError={handleError}
    />
  );
};

async function seedFeatures() {
  console.log('🚀 Starten met aanmaken van features...');
  const transaction = client.transaction();

  for (const feature of features) {
    // Deterministische ID
    const id = `feature-${feature.title.toLowerCase().replace(/[^a-z0-9]/g, '-')}`;

    transaction.createOrReplace({
      _id: id,
      _type: 'feature',
      title: feature.title,
      description: feature.description,
      icon: feature.icon,
    });
    
    console.log(`   📝 Klaargezet: ${feature.title}`);
  }

  try {
    await transaction.commit();
    console.log('✅ Succes! Features zijn aangemaakt in Sanity.');
  } catch (err) {
    console.error('❌ Er ging iets mis:', err.message);
  }
}

const token = process.env.SANITY_TOKEN;

if (!projectId) {
  console.error('❌ Fout: VITE_SANITY_PROJECT_ID ontbreekt in .env');
  process.exit(1);

if (!token) {
  console.error('❌ Fout: SANITY_TOKEN ontbreekt in .env (nodig voor schrijfrechten)');
  process.exit(1);
}

const client = createClient({
  projectId: projectId,
  dataset: 'production',
  token: token, // Schrijfrechten nodig!
  useCdn: false, // We willen direct de laatste data
  apiVersion: '2023-05-03',
});

const links = [
  { name: 'Waarom ik?', href: '#features', order: 1 },
  { name: 'Diensten', href: '#services', order: 2 },
  { name: 'Werk', href: '#portfolio', order: 3 },
  { name: 'Over mij', href: '#about', order: 4 },
  { name: 'Contact', href: '#contact', order: 5 },
];

async function seedNavigation() {
  console.log('🚀 Starten met aanmaken van navigatielinks...');

  // We gebruiken een transactie om alles in één keer te doen
  const transaction = client.transaction();

  for (const link of links) {
    // We maken een deterministische ID (bijv. nav-waarom-ik) zodat we niet dubbel aanmaken
    const id = `nav-${link.name.toLowerCase().replace(/[^a-z0-9]/g, '-')}`;

    transaction.createOrReplace({
      _id: id,
      _type: 'navigationLink',
      name: link.name,
      href: link.href,
      // order veld optioneel als je schema dat ondersteunt, anders wordt het genegeerd
    });
    
    console.log(`   📝 Klaargezet: ${link.name}`);
  }

  try {
    await transaction.commit();
    console.log('✅ Succes! Alle navigatielinks zijn aangemaakt/geüpdatet in Sanity.');
  } catch (err) {
    console.error('❌ Er ging iets mis bij het opslaan:', err.message);
  }
}

seedNavigation();

async function uploadImage(url) {
  try {
    const response = await axios.get(url, { responseType: 'arraybuffer' });
    const buffer = Buffer.from(response.data);
    const asset = await client.assets.upload('image', buffer);
    return asset._id;
  } catch (error) {
    console.error(`Fout bij downloaden/uploaden afbeelding (${url}):`, error.message);
    return null;
  }
}

async function seedPortfolio() {
  console.log('🚀 Starten met seeden van portfolio...');
  const transaction = client.transaction();

  for (const item of portfolioItems) {
    console.log(`   📸 Afbeelding verwerken voor: ${item.title}`);
    
    // Download image from Picsum
    const imageUrl = `https://picsum.photos/id/${item.imageId}/800/600`;
    const assetId = await uploadImage(imageUrl);

    if (!assetId) {
      console.warn(`   ⚠️ Kon geen afbeelding uploaden voor ${item.title}, item wordt overgeslagen.`);
      continue;
    }

    const id = `portfolio-${item.title.toLowerCase().replace(/[^a-z0-9]/g, '-')}`;

    transaction.createOrReplace({
      _id: id,
      _type: 'portfolioItem',
      title: item.title,
      description: item.description,
      techStack: item.techStack,
      status: item.status,
      githubUrl: item.githubUrl,
      liveUrl: item.liveUrl,
      imageUrl: { // LET OP: Dit veld heet 'imageUrl' in je schema!
        _type: 'image',
        asset: {
          _type: 'reference',
          _ref: assetId
        }
      }
    });
    
    console.log(`   📝 Klaargezet: ${item.title}`);
  }

  try {
    await transaction.commit();
    console.log('✅ Succes! Alle portfolio items met afbeeldingen zijn opgeslagen in Sanity.');
  } catch (err) {
    console.error('❌ Er ging iets mis bij de transactie:', err.message);
  }
}

seedPortfolio();
